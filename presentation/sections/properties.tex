\section{Verified Safety Properties}

\begin{frame}{Two Step-Local Safety Properties on \texttt{match}}

Both are stated over the \emph{pre-state} of the match step --- the state before $b$ and $a$ are deactivated.

\begin{block}{Property I --- Best-Order Selection (Price-Time Priority)}
If \texttt{match} produces trade $\mathit{tr}$ from bid $b$ and ask $a$, then in the pre-state:
\[
  \texttt{isBestBid}(b)\ \wedge\ \texttt{isBestAsk}(a).
\]
\emph{The engine cannot print a trade using an out-of-priority order.}
\end{block}

\begin{block}{Property II --- Provenance (Active-on-Fire)}
If \texttt{match} produces trade $\mathit{tr}$ from $b$ and $a$, then in the pre-state:
\[
  b, a \in \texttt{orders}\ \wedge\ \texttt{active}(b)\ \wedge\ \texttt{active}(a).
\]
\emph{Every trade is anchored to real orders that were live when the step fired.}
\end{block}

\end{frame}

\begin{frame}{Property III --- Balance Frame}

\begin{block}{Statement}
For every \texttt{match} step producing trade $\mathit{tr}$ with buyer $b^\star$, seller $a^\star$:
\[
  \forall\, t \notin \{b^\star, a^\star\}.\quad
  \texttt{cash}'[t] = \texttt{cash}[t]\ \wedge\ \texttt{asset}'[t] = \texttt{asset}[t]
\]
\emph{Side-effects are localized to exactly the two participants.}
\end{block}
